class Doughnut {
  cook() {
    print "Fry until golden brown.";
  }
}

class BostonCream > Doughnut {
  cook() {
    super.cook();
    print "Pipe full of custard and coat with chocolate.";
  }
}

BostonCream().cook();

class Math {
  class square(n) {
    return n * n;
  }
}

print Math.square(3); // Prints "9".
